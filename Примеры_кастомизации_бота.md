@tool
def calculate_bmi(weight: float, height: float) -> str:
    """
    Рассчитывает индекс массы тела (BMI).
    
    Args:
        weight: Вес в килограммах
        height: Рост в метрах
    
    Returns:
        str: BMI и его интерпретация
    """
    bmi = weight / (height ** 2)
    
    if bmi < 18.5:
        category = "недостаточный вес"
    elif 18.5 <= bmi < 25:
        category = "нормальный вес"
    elif 25 <= bmi < 30:
        category = "избыточный вес"
    else:
        category = "ожирение"
    
    return f"Ваш BMI: {bmi:.1f} ({category})"

@tool
def create_workout_plan(goal: str, level: str) -> str:
    """
    Создает план тренировок на основе цели и уровня подготовки.
    
    Args:
        goal: Цель (похудение, набор массы, выносливость)
        level: Уровень (начинающий, средний, продвинутый)
    
    Returns:
        str: План тренировок
    """
    plans = {
        "похудение": {
            "начинающий": "3 дня кардио по 30 мин + 2 дня силовые упражнения",
            "средний": "4 дня кардио по 45 мин + 3 дня силовые упражнения",
            "продвинутый": "5 дней интенсивных тренировок (кардио + силовые)"
        },
        # ... другие цели
    }
    
    plan = plans.get(goal.lower(), {}).get(level.lower(), "Не удалось создать план")
    return f"План тренировок для цели '{goal}' (уровень: {level}):\n{plan}"

@tool
def check_python_syntax(code: str) -> str:
    """
    Проверяет синтаксис Python кода.
    
    Args:
        code: Код Python для проверки
    
    Returns:
        str: Результат проверки
    """
    try:
        compile(code, '<string>', 'exec')
        return "✅ Синтаксис кода корректен!"
    except SyntaxError as e:
        return f"❌ Синтаксическая ошибка: {e.msg} в строке {e.lineno}"

@tool
def explain_concept(concept: str) -> str:
    """
    Объясняет концепцию программирования простым языком.
    
    Args:
        concept: Название концепции (например, 'list comprehension')
    
    Returns:
        str: Объяснение с примером
    """
    explanations = {
        "list comprehension": """
        List comprehension - это компактный способ создания списков.
        
        Пример: [x*2 for x in range(5)]
        Результат: [0, 2, 4, 6, 8]
        
        Это то же самое, что:
        result = []
        for x in range(5):
            result.append(x*2)
        """,
        # ... другие концепции
    }
    
    return explanations.get(concept.lower(), f"Концепция '{concept}' пока не добавлена в базу")

@tool
def get_weather(city: str) -> str:
    """
    Получает текущую погоду для города (требует API ключ OpenWeatherMap).
    
    Args:
        city: Название города
    
    Returns:
        str: Информация о погоде
    """
    import requests
    
    API_KEY = os.getenv("OPENWEATHER_API_KEY")
    url = f"http://api.openweathermap.org/data/2.5/weather?q={city}&appid={API_KEY}&units=metric&lang=ru"
    
    try:
        response = requests.get(url)
        data = response.json()
        
        temp = data['main']['temp']
        description = data['weather'][0]['description']
        
        return f"🌡️ Погода в {city}: {temp}°C, {description}"
    except Exception as e:
        return f"Не удалось получить погоду для {city}"

### 2. Работа с курсами валют

@tool
def get_exchange_rate(from_currency: str, to_currency: str) -> str:
    """
    Получает курс обмена валют.
    
    Args:
        from_currency: Исходная валюта (например, USD)
        to_currency: Целевая валюта (например, EUR)
    
    Returns:
        str: Курс обмена
    """
    import requests
    
    url = f"https://api.exchangerate-api.com/v4/latest/{from_currency.upper()}"
    
    try:
        response = requests.get(url)
        data = response.json()
        rate = data['rates'].get(to_currency.upper())
        
        if rate:
            return f"💱 1 {from_currency.upper()} = {rate:.2f} {to_currency.upper()}"
        else:
            return f"Валюта {to_currency} не найдена"
    except Exception as e:
        return f"Ошибка получения курса валют: {e}"
```

def save_user_note(user_id: int, note: str) -> str:

        user_id: ID пользователя

    conn.close()

@tool
def get_user_notes(user_id: int) -> str:
    """
    Получает все заметки пользователя.
    
    Args:
        user_id: ID пользователя
    
    Returns:
        str: Список заметок
    """
    import sqlite3
    
    conn = sqlite3.connect('user_notes.db')
    cursor = conn.cursor()
    
    cursor.execute('SELECT note, created_at FROM notes WHERE user_id = ?', (user_id,))
    notes = cursor.fetchall()
    
    conn.close()

        result = "📝 Ваши заметки:\n\n"

import sqlite3

# Инициализация БД
def init_db():
    conn = sqlite3.connect('users.db')
    cursor = conn.cursor()
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS users (
            user_id INTEGER PRIMARY KEY,
            username TEXT,
            registered_at TIMESTAMP,
            preferences TEXT
        )
    ''')
    conn.commit()
    conn.close()

# При старте бота
init_db()

@dp.message(Command("start"))
async def start_with_registration(message: Message):
    user_id = message.from_user.id
    username = message.from_user.username or "Unknown"
    
    conn = sqlite3.connect('users.db')
    cursor = conn.cursor()
    
    # Проверяем, зарегистрирован ли пользователь
    cursor.execute('SELECT user_id FROM users WHERE user_id = ?', (user_id,))
    user = cursor.fetchone()
    
    if not user:
        # Регистрируем нового пользователя
        from datetime import datetime
        cursor.execute(
            'INSERT INTO users (user_id, username, registered_at) VALUES (?, ?, ?)',
            (user_id, username, datetime.now())
        )
        conn.commit()
        await message.reply(f"👋 Добро пожаловать, {username}! Вы успешно зарегистрированы.")
    else:
        await message.reply(f"С возвращением, {username}!")
    
    conn.close()
```

### 3. Бот с расписанием (напоминания)

```python
from datetime import datetime, timedelta
import asyncio

# Список напоминаний (в реальном боте лучше использовать БД)
reminders = []

@dp.message(Command("remind"))
async def set_reminder(message: Message):
    # Пример: /remind 10 Купить молоко
    # Напомнит через 10 минут
    
    parts = message.text.split(' ', 2)
    if len(parts) < 3:
        await message.reply("Используйте: /remind <минуты> <текст напоминания>")
        return
    
    try:
        minutes = int(parts[1])
        reminder_text = parts[2]
        remind_at = datetime.now() + timedelta(minutes=minutes)
        
        reminders.append({
            'user_id': message.from_user.id,
            'text': reminder_text,
            'time': remind_at
        })
        
        await message.reply(f"✅ Напоминание установлено на {remind_at.strftime('%H:%M')}")
        
        # Запускаем задачу напоминания
        asyncio.create_task(send_reminder(message.from_user.id, reminder_text, minutes))
        
    except ValueError:
        await message.reply("Укажите корректное количество минут")

async def send_reminder(user_id: int, text: str, minutes: int):
    """Отправляет напоминание через указанное время"""
    await asyncio.sleep(minutes * 60)
    await bot.send_message(user_id, f"⏰ Напоминание: {text}")
```

### 3. Middleware для аналитики
```python
from aiogram import BaseMiddleware

class AnalyticsMiddleware(BaseMiddleware):
    async def __call__(self, handler, event, data):
        # Логика до обработки
        user_id = event.from_user.id if hasattr(event, 'from_user') else None
        logger.info(f"Входящее событие от пользователя {user_id}")
        
        # Вызываем обработчик
        result = await handler(event, data)
        
        # Логика после обработки
        logger.info(f"Событие обработано для пользователя {user_id}")
        
        return result

# Регистрация middleware
dp.message.middleware(AnalyticsMiddleware())
```
